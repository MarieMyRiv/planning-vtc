function layoutEvents(evs){

  if(!evs.length)return[];

  const sorted=[...evs].sort((a,b)=>a.start-b.start);

  const cols=[];const result=[];

  for(const ev of sorted){

    let placed=false;

    for(let c=0;c<cols.length;c++){

      const last=cols[c][cols[c].length-1];

      if(ev.start>=last.end){cols[c].push(ev);result.push({ev,col:c});placed=true;break;}

    }

    if(!placed){cols.push([ev]);result.push({ev,col:cols.length-1});}

  }

  return result.map(item=>{

    const active=result.filter(x=>x.ev.start<item.ev.end&&x.ev.end>item.ev.start);

    return{...item,totalCols:Math.max(...active.map(x=>x.col))+1};

  });

}

function renderDay(){

  const d=selDs();const dayEvs=evDay(d);

  const totalH=24*HOUR_PX;const now=new Date();

  const isToday=d===todayDs();const nowM=now.getHours()*60+now.getMinutes();

  const groups=[];

  DRIVERS.forEach(dr=>{const drEvs=dayEvs.filter(e=>e.driver===dr.id);if(drEvs.length)groups.push({dr,layout:layoutEvents(drEvs)});});

  let h=`<div style="display:flex;height:${totalH}px;min-width:max-content">`;

  h+=timeColHTML(totalH);

  if(!groups.length){

    h+=`<div style="flex:1;min-width:${MIN_COL_W}px;position:relative;height:${totalH}px;cursor:crosshair" onclick="openNewAt(event,'d1','${d}')">${gridLines()}${isToday?`<div class="now-line" style="top:${nowM*PX}px"><div class="now-dot"></div></div>`:''}</div>`;

  } else {

    groups.forEach(g=>{

      const nSub=Math.max(...g.layout.map(x=>x.totalCols));

      const colW=MIN_COL_W*nSub;

      h+=`<div style="width:${colW}px;min-width:${colW}px;border-right:1px solid #e5e7eb;border-top:3px solid ${g.dr.hdr};position:relative;height:${totalH}px;cursor:crosshair" onclick="openNewAt(event,'${g.dr.id}','${d}')">`;

      g.layout.forEach(item=>{h+=evHTML(item.ev,g.dr,item.col,item.totalCols,colW);});

      h+=`</div>`;

    });

  }

  h+=`</div>`;

  const va=document.getElementById('view-area');va.innerHTML=h;

  setTimeout(()=>{va.scrollTop=isToday?Math.max(0,nowM*PX-120):8*HOUR_PX;},50);

function openEdit(id){

  const ev=events.find(e=>e.id===id);if(!ev)return;

  resetModal();editId=id;

  document.getElementById('modal-title').textContent='Modifier la course';

  document.getElementById('btn-del').style.display='';

  document.getElementById('f-driver').innerHTML=driverOpts(ev.driver);

  document.getElementById('f-type').value=ev.type||'TRSF IN';

  document.getElementById('f-date').value=ev.date;

  document.getElementById('f-start').value=ev.start;

  document.getElementById('f-end').value=ev.end;

  ['pickup','dropoff','client','pax','tel','agence','note'].forEach(k=>{document.getElementById('f-'+k).value=ev[k]||'';});

  if(ev.admin){

    document.getElementById('f-achat').value=ev.admin.achat||'';

    document.getElementById('f-vente').value=ev.admin.vente||'';

    calcMarge();

    const rb=document.querySelector(`input[name="encaiss"][value="${ev.admin.encaiss}"]`);if(rb)rb.checked=true;

    document.getElementById('f-date-enc').value=ev.admin.dateEnc||'';

    document.getElementById('f-factu-my').value=ev.admin.factuMy||'';

    document.getElementById('f-st-factu').value=ev.admin.factuSt||'';

    document.getElementById('f-st-date').value=ev.admin.dateSt||'';

    const rb2=document.querySelector(`input[name="st-paie"][value="${ev.admin.paieSt}"]`);if(rb2)rb2.checked=true;

    document.getElementById('f-notes-admin').value=ev.admin.notes||'';

  }

  document.getElementById('modal-bg').classList.add('open');

}

function calcMarge(){

  const a=parseFloat(document.getElementById('f-achat').value)||0;

  const v=parseFloat(document.getElementById('f-vente').value)||0;

  const m=v-a;const el=document.getElementById('f-marge');

  el.textContent=m.toFixed(0)+' €';el.style.color=m>=0?'#059669':'#dc2626';

}

  const ev={

    id:editId||('e'+Date.now()),

    driver:document.getElementById('f-driver').value,

    type:document.getElementById('f-type').value,

    date:evDate,start,end,

    pickup:document.getElementById('f-pickup').value,

    pax:document.getElementById('f-pax').value,

    admin:{

      encaiss,dateEnc:document.getElementById('f-date-enc').value,

  await saveToSheets(ev);

}
